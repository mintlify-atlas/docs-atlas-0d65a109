## Core types

### Document

Base interface for all documents stored in the database.

```typescript
export interface Document {
  _id: string;
  [key: string]: unknown;
}
```

<ParamField path="_id" type="string" required>
  Unique identifier for the document (auto-generated on insert)
</ParamField>

<ParamField path="[key: string]" type="unknown">
  Additional fields with any values
</ParamField>

**Example**

```typescript
import { Document } from '@sohzm/jasonisnthappy';

interface User extends Document {
  _id: string;
  name: string;
  email: string;
  age: number;
  tags: string[];
}
```

## Configuration types

### DatabaseOptions

Configuration options for opening a database.

```typescript
export interface DatabaseOptions {
  cacheSize?: number;
  autoCheckpointThreshold?: number;
  filePermissions?: number;
  readOnly?: boolean;
  maxBulkOperations?: number;
  maxDocumentSize?: number;
  maxRequestBodySize?: number;
}
```

<ParamField path="cacheSize" type="number">
  Size of the page cache in bytes. Larger cache improves read performance.
</ParamField>

<ParamField path="autoCheckpointThreshold" type="number">
  Number of pages written before automatic checkpoint triggers.
</ParamField>

<ParamField path="filePermissions" type="number">
  Unix file permissions for the database file (e.g., `0o644`).
</ParamField>

<ParamField path="readOnly" type="boolean">
  Open database in read-only mode if `true`.
</ParamField>

<ParamField path="maxBulkOperations" type="number">
  Maximum number of operations allowed in a single bulk write.
</ParamField>

<ParamField path="maxDocumentSize" type="number">
  Maximum size of a single document in bytes.
</ParamField>

<ParamField path="maxRequestBodySize" type="number">
  Maximum request body size for web server (if enabled).
</ParamField>

**Example**

```typescript
const options: DatabaseOptions = {
  cacheSize: 1024 * 1024 * 100, // 100MB
  autoCheckpointThreshold: 1000,
  readOnly: false,
  maxBulkOperations: 10000,
  maxDocumentSize: 1024 * 1024 * 10 // 10MB
};

const db = Database.openWithOptions('./data.db', options);
```

### TransactionConfig

Configuration for transaction retry behavior.

```typescript
export interface TransactionConfig {
  maxRetries?: number;
  retryBackoffBaseMs?: number;
  maxRetryBackoffMs?: number;
}
```

<ParamField path="maxRetries" type="number">
  Maximum number of retry attempts for conflicted transactions.
</ParamField>

<ParamField path="retryBackoffBaseMs" type="number">
  Base delay in milliseconds for exponential backoff.
</ParamField>

<ParamField path="maxRetryBackoffMs" type="number">
  Maximum delay in milliseconds between retries.
</ParamField>

**Example**

```typescript
const config: TransactionConfig = {
  maxRetries: 5,
  retryBackoffBaseMs: 10,
  maxRetryBackoffMs: 1000
};

db.setTransactionConfig(config);
```

</ParamField>

  JMESPath filter to match documents
</ParamField>

**Example**

```typescript
import { BulkOperation } from '@sohzm/jasonisnthappy';

const operations: BulkOperation<Omit<User, '_id'>>[] = [
  {
    op: 'insert',
    doc: { name: 'Alice', email: 'alice@example.com', age: 30 }
  },
  {
    op: 'update_one',
    filter: '[?email == `bob@example.com`]',
    update: { age: 26 }
  },
  {
    op: 'update_many',
    filter: '[?age < `18`]',
    update: { status: 'minor' }
  },
  {
    op: 'delete_one',
    filter: '[?email == `old@example.com`]'
  },
  {
    op: 'delete_many',
    filter: '[?active == `false`]'
  }
];

const result = users.bulkWrite(operations);
```

## Aggregation types

### AggregationStage

Aggregation pipeline stage.

```typescript
export interface AggregationStage {
  match?: string;
  group_by?: string;
  count?: string;
  sum?: { field: string; output: string };
  avg?: { field: string; output: string };
  min?: { field: string; output: string };
  max?: { field: string; output: string };
  sort?: { field: string; asc?: boolean };
  limit?: number;
  skip?: number;
  project?: string[];
  exclude?: string[];
}
```

<ParamField path="match" type="string">
  JMESPath filter to match documents
</ParamField>

<ParamField path="group_by" type="string">
  Field name to group by
</ParamField>

<ParamField path="count" type="string">
  Output field name for count
</ParamField>

<ParamField path="sum" type="{ field: string; output: string }">
  Sum aggregation: `field` to sum, `output` field name for result
</ParamField>

<ParamField path="avg" type="{ field: string; output: string }">
  Average aggregation: `field` to average, `output` field name for result
</ParamField>

<ParamField path="min" type="{ field: string; output: string }">
  Minimum aggregation: `field` to find min, `output` field name for result
</ParamField>

<ParamField path="max" type="{ field: string; output: string }">
  Maximum aggregation: `field` to find max, `output` field name for result
</ParamField>

<ParamField path="sort" type="{ field: string; asc?: boolean }">
  Sort by field: `field` to sort by, `asc` for ascending (default: true)
</ParamField>

<ParamField path="limit" type="number">
  Maximum number of documents to return
</ParamField>

<ParamField path="skip" type="number">
  Number of documents to skip
</ParamField>

<ParamField path="project" type="string[]">
  Fields to include in output
</ParamField>

<ParamField path="exclude" type="string[]">
  Fields to exclude from output
</ParamField>

**Example**

```typescript
interface AgeStats {
  age: number;
  count: number;
  avg_score: number;
}

const stats = users.aggregate<AgeStats>([
  { match: '[?active == `true`]' },
  { group_by: 'age' },
  { count: 'count' },
  { avg: { field: 'score', output: 'avg_score' } },
  { sort: { field: 'age', asc: true } },
  { limit: 10 }
]);
```

## Watch types

### ChangeOperation

Type of change operation.

```typescript
export type ChangeOperation = 'insert' | 'update' | 'delete';
```

### WatchCallback

Callback function for watching collection changes.

```typescript
export type WatchCallback<T> = (
  operation: ChangeOperation,
  docId: string,
  document: T | null
) => void;
```

<ParamField path="operation" type="ChangeOperation" required>
  Type of operation: `'insert'`, `'update'`, or `'delete'`
</ParamField>

<ParamField path="docId" type="string" required>
  Document ID that changed
</ParamField>

<ParamField path="document" type="T | null" required>
  The document data (null for delete operations)
</ParamField>

**Example**

```typescript
const callback: WatchCallback<User> = (op, docId, doc) => {
  switch (op) {
    case 'insert':
      console.log(`New user: ${doc.name}`);
      break;
    case 'update':
      console.log(`Updated user: ${doc.name}`);
      break;
    case 'delete':
      console.log(`Deleted user: ${docId}`);
      break;
  }
};

const handle = users.watch(undefined, callback);
```
